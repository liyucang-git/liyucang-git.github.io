# 常见排序算法

## 直接插入排序

```
function insertSort(arr) {
  for (let i = 1; i <= arr.length - 1; i++) {
    const temp = arr[i];
    let j = i - 1;
    for (; (j >= 0) && (arr[j] > temp); j--) {
      arr[j + 1] = arr[j];
    }
    arr[j + 1] = temp;
  }
  return arr;
}
```

## 希尔排序：

```
function shierSort(arr) {
  let step = Math.floor(arr.length / 2);
  for (; step >= 1; step = Math.floor(step / 2)) {
    for (let i = step; i <= arr.length - 1; i++) {
      const temp = arr[i];
      let j = i - step;
      for (; (j >= 0) && (arr[j] > temp); j = j - step) {
        arr[j + step] = arr[j];
      }
      arr[j + step] = temp;
    }
  }
  return arr;
}
```

## 选择排序

```
function selectSort(arr) {
  for (let i = 0; i <= arr.length - 2; i++) {
    let min = i;
    for (let j = i + 1; j <= arr.length -1; j++) {
      if (arr[j] < arr[min]) {
        min = j;
      }
    }
    if (min !== i) {
      swap(arr, i, min);
    }
  }
  return arr;
}

function swap(arr, indexA, indexB) {
  const temp = arr[indexA];
  arr[indexA] = arr[indexB];
  arr[indexB] = temp;
}
```

## 堆排序

function adjustHip(arr, index, maxIndex) {
  let k = index;
  while (k <= maxIndex) {
    let left = k * 2 + 1;
    let right = k * 2 + 2;
    if (left > maxIndex) {
      break;
    }

    let maxChild = left;
    if ((right <= maxIndex) && (arr[right] >= arr[left])) {
      maxChild = right;
    }
    if (arr[i] > arr[k]) {
      swap(arr, maxChild, k);
      k = maxChild;
    } else {
      break;
    }
  }
}

function buildHip(arr) {
  let i = Math.floor((arr.length - 1) / 2);
  for (; i >= 0; i--) {
    adjustHip(arr, i, arr.length - 1);
  }
}

function hipSort(arr) {
  buildHip(arr);
  for (let i = arr.length - 1; i >= 1;) {
    swap(arr, 0, i);
    adjustHip(arr, 0, --i);
  }
  return arr;
}

function swap(arr, indexA, indexB) {
  const temp = arr[indexA];
  arr[indexA] = arr[indexB];
  arr[indexB] = temp;
}

function popSort(arr) {
  for (let i = arr.length -2; i >= 0; i--) {
    for (let j = 0; j <= i; j++) {
      if (arr[j] > arr[j + 1]) {
        swap(arr, j, j + 1);
      }
    }
  }
  return arr;
}

## 快排

```
function quickSort(arr, start = 0, end = arr.length - 1) {
  if (start < end) {
    const index = geneIndex(arr, start, end);
    quick(arr, start, index -1);
    quick(arr, index + 1, end);
  }
  return arr;
}

function geneIndex(arr, start, end) {
  let j = start - 1;
  const mid = arr[end];
  for (let i = start; i <= end - 1; i++) {
    if (arr[i] < mid) {
      if (j + 1 < i) {
        const tmp = arr[i];
        arr[i] = arr[j + 1];
        arr[j + 1] = tmp;
      }
      j++;
    }
  }

  if (j + 1 !== end) {
    const tmp = arr[end];
    arr[end] = arr[j + 1];
    arr[j + 1] = tmp;
  }

  return j + 1;
}
```

归并：

```
function second(arr, start = 0, end = arr.length - 1) {
  if (start >= end) {
    return arr;
  }

  const mid = ((end - start) >> 1) + start;
  second(arr, start, mid);
  second(arr, mid + 1, end);

  const copyArr = [];

  for (let i = start; i <= end; i++) {
    copyArr[i] = arr[i];
  }

  let i = start, j = mid + 1, k = start;

  while ((i <= mid) && (j <= end)) {
    if (copyArr[i] <= copyArr[j]) {
      arr[k++] = copyArr[i++];
    } else {
      arr[k++] = copyArr[j++];
    }
  }

  while (i <= mid) {
    arr[k++] = copyArr[i++];
  }

  while (j <= end) {
    arr[k++] = copyArr[j++];
  }

  return arr;
}
```

function midSearch(arr, val) {
  if (arr.length <= 1) {
    return arr;
  }

  let low = 0;
  let height = arr.length - 1;
  while (low <= height) {
    const mid = ((height - low) >> 1) + low;
    if (arr[mid] === val) {
      return mid;
    }

    if (arr[mid] < val) {
      low = mid + 1;
    } else {
      height = mid - 1;
    }
  }

  return;
}
