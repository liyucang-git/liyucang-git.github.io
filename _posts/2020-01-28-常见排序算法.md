# 常见排序算法

直接插入排序：

```
function insert(arr) {
  for (let i = 1; i <= arr.length - 1; i++) {
    const val = arr[i];
    let j = i - 1;
    for (; (j >= 0) && (arr[j] > val); j--) {
      arr[j + 1] = arr[j];
    }
    arr[j + 1] = val;
  }
  return arr;
}
```

选择排序：

```
function select(arr) {
  for (let i = 0; i <= arr.length - 2; i++) {
    let min = i;
    for (let j = i + 1; j <= arr.length -1; j++) {
      if (arr[j] < arr[min]) {
        min = j;
      }
    }
    if (min !== i) {
      const tmp = arr[i];
      arr[i] = arr[min];
      arr[min] = tmp;
    }
  }
  return arr;
}
```

冒泡排序：

```
function pop(arr) {
  for (let i = arr.length -1; i >= 1; i--) {
    for (let j = 0; j + 1 <= i; j++) {
      if (arr[j] > arr[j + 1]) {
        const tmp = arr[j + 1];
        arr[j + 1] = arr[j];
        arr[j] = tmp;
      }
    }
  }
  return arr;
}
```

快排：

```
function quick(arr, start = 0, end = arr.length - 1) {
  if (start < end) {
    const index = geneIndex(arr, start, end);
    quick(arr, start, index -1);
    quick(arr, index + 1, end);
  }

  return arr;
}

function geneIndex(arr, start, end) {
  let j = start - 1;
  const mid = arr[end];
  for (let i = start; i <= end - 1; i++) {
    if (arr[i] < mid) {
      if (j + 1 < i) {
        const tmp = arr[i];
        arr[i] = arr[j + 1];
        arr[j + 1] = tmp;
      }
      j++;
    }
  }

  if (j + 1 !== end) {
    const tmp = arr[end];
    arr[end] = arr[j + 1];
    arr[j + 1] = tmp;
  }

  return j + 1;
}
```

归并：

```
function second(arr, start = 0, end = arr.length - 1) {
  if (start >= end) {
    return arr;
  }

  const mid = ((end - start) >> 1) + start;
  second(arr, start, mid);
  second(arr, mid + 1, end);

  const copyArr = [];

  for (let i = start; i <= end; i++) {
    copyArr[i] = arr[i];
  }

  let i = start, j = mid + 1, k = start;

  while ((i <= mid) && (j <= end)) {
    if (copyArr[i] <= copyArr[j]) {
      arr[k++] = copyArr[i++];
    } else {
      arr[k++] = copyArr[j++];
    }
  }

  while (i <= mid) {
    arr[k++] = copyArr[i++];
  }

  while (j <= end) {
    arr[k++] = copyArr[j++];
  }

  return arr;
}
```

二分查找：

```
function midSearch(arr, val) {
  if (arr.length <= 1) {
    return arr;
  }

  let low = 0;
  let height = arr.length - 1;
  while (low <= height) {
    const mid = ((height - low) >> 1) + low;
    if (arr[mid] === val) {
      return mid;
    }

    if (arr[mid] < val) {
      low = mid + 1;
    } else {
      height = mid - 1;
    }
  }

  return;
}
```
