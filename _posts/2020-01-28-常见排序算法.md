# 常见排序算法

## 直接插入排序

```
function insertSort(arr) {
  for (let i = 1; i <= arr.length - 1; i++) {
    const temp = arr[i];
    let j = i - 1;
    for (; (j >= 0) && (arr[j] > temp); j--) {
      arr[j + 1] = arr[j];
    }
    arr[j + 1] = temp;
  }
  return arr;
}
```

## 希尔排序：

```
function shierSort(arr) {
  let step = Math.floor(arr.length / 2);
  for (; step >= 1; step = Math.floor(step / 2)) {
    for (let i = step; i <= arr.length - 1; i++) {
      const temp = arr[i];
      let j = i - step;
      for (; (j >= 0) && (arr[j] > temp); j = j - step) {
        arr[j + step] = arr[j];
      }
      arr[j + step] = temp;
    }
  }
  return arr;
}
```

## 选择排序

```
function selectSort(arr) {
  for (let i = 0; i <= arr.length - 2; i++) {
    let min = i;
    for (let j = i + 1; j <= arr.length -1; j++) {
      if (arr[j] < arr[min]) {
        min = j;
      }
    }
    if (min !== i) {
      swap(arr, i, min);
    }
  }
  return arr;
}

function swap(arr, indexA, indexB) {
  const temp = arr[indexA];
  arr[indexA] = arr[indexB];
  arr[indexB] = temp;
}
```

## 快排

```
function quickSort1(arr, start = 0, end = arr.length - 1) {
  if (start < end) {
    const index = geneIndex(arr, start, end);
    quickSort(arr, start, index -1);
    quickSort(arr, index + 1, end);
  }
  return arr;
}

function quickSort2(arr) {
  if (arr.length <= 1) {
    return arr;
  }

  const stack = [];
  stack.push(0);
  stack.push(arr.length - 1);

  while (stack.length) {
    const start = stack.shift();
    const end = stack.shift();
    if (start < end) {
      const index = geneIndex(arr, start, end);
      stack.push(start, index -1);
      stack.push(index + 1, end);
    }
  }
  return arr;
}

function geneIndex(arr, start, end) {
  const index = Math.floor(end - start) + start;
  swap(arr, index, end);

  let j = start - 1;
  const mid = arr[end];
  for (let i = start; i <= end - 1; i++) {
    if (arr[i] < mid) {
      if (j + 1 < i) {
        swap(arr, i, j + 1);
      }
      j++;
    }
  }
  if (j + 1 !== end) {
    swap(arr, j + 1, end);
  }
  return j + 1;
}

function swap(arr, indexA, indexB) {
  const temp = arr[indexA];
  arr[indexA] = arr[indexB];
  arr[indexB] = temp;
}
```

## 归并排序

```
function secondSort(arr, start = 0, end = arr.length - 1) {
  if (start >= end) {
    return arr;
  }

  const mid = Math.floor((end - start) / 2) + start;
  secondSort(arr, start, mid);
  secondSort(arr, mid + 1, end);

  const copyArr = [];
  for (let i = start; i <= end; i++) {
    copyArr[i] = arr[i];
  }

  let i = start, j = mid + 1, k = start;
  while ((i <= mid) && (j <= end)) {
    if (copyArr[i] <= copyArr[j]) {
      arr[k++] = copyArr[i++];
    } else {
      arr[k++] = copyArr[j++];
    }
  }
  while (i <= mid) {
    arr[k++] = copyArr[i++];
  }
  while (j <= end) {
    arr[k++] = copyArr[j++];
  }
  return arr;
}
```

## 计数排序

```
function countSort(arr) {
  let min = arr[0];
  let max = arr[0];
  for (let i = 1; i <= arr.length -1; i++) {
    if (arr[i] > max) {
      max = arr[i];
    } else if (arr[i] < min) {
      min = arr[i];
    }
  }

  const bucket = new Array(max - min + 1);
  for (let i = 0; i <= arr.length -1; i++) {
    const key = arr[i] - min;
    if (!bucket[key]) {
      bucket[key] = 0;
    }
    bucket[key]++;
  }

  let k = 0;
  for (let i = 0; i <= bucket.length -1; i++) {
    if (bucket[i]) {
      while (bucket[i]) {
        arr[k++] = i + min;
        bucket[i]--;
      }
    }
  }
  return arr;
}
```

## 桶排序

```
function bucketSort(arr, bucketSize) {
  if (arr.length <= 1) {
    return arr;
  }

  let min = arr[0];
  let max = arr[0];
  for (let i = 1; i <= arr.length -1; i++) {
    if (arr[i] > max) {
      max = arr[i];
    } else if (arr[i] < min) {
      min = arr[i];
    }
  }

  if (min === max) {
    return arr;
  }

  const bucketLen = Math.ceil((max - min + 1) / bucketSize);
  if (bucketLen === 1) {
    return bucketSort(arr, bucketSize - 1);
  }

  const bucketArr = new Array(bucketLen);
  for (let i = 0; i <= bucketArr.length -1; i++) {
    bucketArr[i] = [];
  }

  for (let i = 0; i <= arr.length - 1; i++) {
    const bucketIndex = Math.ceil((arr[i] - min + 1) / bucketSize) - 1;
    bucketArr[bucketIndex].push(arr[i]);
  }

  const resultArr = [];
  for (let i = 0; i <= bucketArr.length -1; i++) {
    const sortArr = bucketSort(bucketArr[i], bucketSize)
    resultArr.push(...sortArr);
  }
  return resultArr;
}
```

## 二分查找

```
function midSearch(arr, val) {
  if (!arr.length) {
    return false;
  }

  let low = 0;
  let height = arr.length - 1;
  while (low <= height) {
    const mid = ((height - low) >> 1) + low;
    if (arr[mid] === val) {
      return mid;
    }

    if (arr[mid] < val) {
      low = mid + 1;
    } else {
      height = mid - 1;
    }
  }

  return false;
}
```
