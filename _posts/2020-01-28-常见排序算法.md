# 常见排序算法

直接插入排序：

```
function insert(arr) {
  for (let i = 1; i <= arr.length - 1; i++) {
    const val = arr[i];
    if (arr[i] < arr[i - 1]) {
      let j = i - 1;
      for (; (j >= 0) && (arr[j] > val); j--) {
        arr[j + 1] = arr[j];
      }
      arr[j + 1] = val;
    }
  }
  return arr;
}
```

选择排序：

```
function select(arr) {
  for (let i = 0; i <= arr.length - 2; i++) {
    let min = i;
    for (let j = i + 1; j <= arr.length -1; j++) {
      if (arr[j] < arr[min]) {
        min = j;
      }
    }
    if (min !== i) {
      const tmp = arr[i];
      arr[i] = arr[min];
      arr[min] = tmp;
    }
  }
  return arr;
}
```

冒泡排序：

```
function pop(arr) {
  for (let i = arr.length -1; i >= 1; i--) {
    for (let j = 0; j + 1 <= i; j++) {
      if (arr[j] > arr[j + 1]) {
        const tmp = arr[j + 1];
        arr[j + 1] = arr[j];
        arr[j] = tmp;
      }
    }
  }
  return arr;
}
```

快排：

```
function quick(arr, start = 0, end = arr.length - 1) {
  const index = geneIndex(arr, start, end);
  if (index - 1 > start) {
    quick(arr, start, index -1);
  }
  if (index + 1 < end) {
    quick(arr, index + 1, end);
  }
  return arr;
}

function geneIndex(arr, start, end) {
  if (start === end) {
    return start;
  }

  let j = start - 1;
  const mid = arr[end];
  for (let i = start; i <= end - 1; i++) {
    if (arr[i] < mid) {
      if (j + 1 < i) {
        const tmp = arr[i];
        arr[i] = arr[j + 1];
        arr[j + 1] = tmp;
      }
      j++;
    }
  }

  if (j + 1 !== end) {
    const tmp = arr[end];
    arr[end] = arr[j + 1];
    arr[j + 1] = tmp;
  }

  return j + 1;
}
```

归并：

```
function second(arr, start = 0, end = arr.length - 1) {
  if (start === end) {
    return arr;
  }

  const mid = ((end - start) >> 1) + start;
  second(arr, start, mid);
  second(arr, mid + 1, end);

  const arr1 = [];
  const arr2 = [];

  for (let i = start, j = 0; i <= mid;) {
    arr1[j] = arr[i];
    i++;
    j++;
  }

  for (let i = mid + 1, j = 0; i <= end;) {
    arr2[j] = arr[i];
    i++;
    j++;
  }

  let i = 0, j = 0, k = 0;
  for (; (i <= arr1.length -1) && (j <= arr2.length -1);) {
    if (arr1[i] <= arr2[j]) {
      arr[start + k] = arr1[i];
      i++;
      k++;
    } else {
      arr[start + k] = arr2[j];
      j++;
      k++;
    }
  }

  for (; i <= arr1.length - 1; i++) {
    arr[start + k] = arr1[i];
    k++;
  }

  for (; j <= arr2.length - 1; j++) {
    arr[start + k] = arr2[j];
    k++;
  }

  return arr;
}
```
